149ComputeAdtQcMetricsResults<Sum_, Detected_> compute_adt_qc_metrics(

153{

155 ComputeAdtQcMetricsBuffers<Sum_, Detected_> x;

172 compute_adt_qc_metrics(mat, subsets, x, options);

204template<typename Float_, class Host_, typename Sum_, typename Detected_, typename BlockSource_>
205void adt_populate(Host_& host, size_t n, const ComputeAdtQcMetricsBuffers<Sum_, Detected_>& res, BlockSource_ block, const ComputeAdtQcFiltersOptions& options) {
206 constexpr bool unblocked = std::is_same<BlockSource_, bool>::value;
207 auto buffer = [&]() {
208 if constexpr(unblocked) {
209 return std::vector<Float_>(n);
210 } else {
211 return FindMedianMadWorkspace<Float_, size_t>(n, block);
212 }
213 }();
214
215 {
216 ChooseFilterThresholdsOptions opts;
217 opts.num_mads = options.detected_num_mads;
218 opts.log = true;
219 opts.upper = false;
220 opts.min_diff = -std::log(1 - options.detected_min_drop);
221 host.get_detected() = [&]() {
222 if constexpr(unblocked) {
223 return choose_filter_thresholds(n, res.detected, buffer.data(), opts).lower;
224 } else {
225 return internal::strip_threshold<true>(choose_filter_thresholds_blocked(n, res.detected, block, &buffer, opts));
226 }
227 }();
228 }
229
230 {
231 size_t nsubsets = res.subset_sum.size();
232 host.get_subset_sum().resize(nsubsets);
233
234 ChooseFilterThresholdsOptions opts;
235 opts.num_mads = options.subset_sum_num_mads;
236 opts.log = true;
237 opts.lower = false;
238
239 for (size_t s = 0; s < nsubsets; ++s) {
240 auto sub = res.subset_sum[s];
241 host.get_subset_sum()[s] = [&]() {
242 if constexpr(unblocked) {
243 return choose_filter_thresholds(n, sub, buffer.data(), opts).upper;
244 } else {
245 return internal::strip_threshold<false>(choose_filter_thresholds_blocked(n, sub, block, &buffer, opts));
246 }
247 }();
248 }
249 }

252template<class Host_, typename Sum_, typename Detected_, typename BlockSource_, typename Output_>
253void adt_filter(const Host_& host, size_t n, const ComputeAdtQcMetricsBuffers<Sum_, Detected_>& metrics, BlockSource_ block, Output_* output) {
254 constexpr bool unblocked = std::is_same<BlockSource_, bool>::value;
255 std::fill_n(output, n, 1);
256
257 for (size_t i = 0; i < n; ++i) {
258 auto thresh = [&]() {
259 if constexpr(unblocked) {
260 return host.get_detected();
261 } else {
262 return host.get_detected()[block[i]];
263 }
264 }();
265 output[i] = output[i] && (metrics.detected[i] >= thresh);
266 }
267
268 size_t nsubsets = metrics.subset_sum.size();
269 for (size_t s = 0; s < nsubsets; ++s) {
270 auto sub = metrics.subset_sum[s];
271 const auto& sthresh = host.get_subset_sum()[s];
272 for (size_t i = 0; i < n; ++i) {
273 auto thresh = [&]() {
274 if constexpr(unblocked) {
275 return sthresh;
276 } else {
277 return sthresh[block[i]];
278 }
279 }();
280 output[i] = output[i] && (sub[i] <= thresh);
281 }
282 }
283}
