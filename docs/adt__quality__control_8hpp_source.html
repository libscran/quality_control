166template<typename Sum_ = double, typename Detected_ = int, typename Value_ = double, typename Index_ = int, typename Subset_ = const uint8_t*>
167MetricsResults<Sum_, Detected_> compute_metrics(const tatami::Matrix<Value_, Index_>* mat, const std::vector<Subset_>& subsets, const MetricsOptions& options) {
168 auto NC = mat->ncol();
169 MetricsBuffers<Sum_, Detected_> x;
170 MetricsResults<Sum_, Detected_> output;
171
172 output.sum.resize(NC);
173 x.sum = output.sum.data();
174
175 output.detected.resize(NC);
176 x.detected = output.detected.data();
177
178 size_t nsubsets = subsets.size();
179 x.subset_sum.resize(nsubsets);
180 output.subset_sum.resize(nsubsets);
181 for (size_t s = 0; s < nsubsets; ++s) {
182 output.subset_sum[s].resize(NC);
183 x.subset_sum[s] = output.subset_sum[s].data();
184 }
185
186 compute_metrics(mat, subsets, x, options);

218template<typename Float_, class Host_, typename Sum_, typename Detected_, typename BlockSource_>
219void populate(Host_& host, size_t n, const MetricsBuffers<Sum_, Detected_>& res, BlockSource_ block, const FiltersOptions& options) {
220 constexpr bool unblocked = std::is_same<BlockSource_, bool>::value;
221 auto buffer = [&]() {
222 if constexpr(unblocked) {
223 return std::vector<Float_>(n);
224 } else {
225 return find_median_mad::Workspace<Float_, size_t>(n, block);
226 }
227 }();
228
229 {
230 choose_filter_thresholds::Options opts;
231 opts.num_mads = options.detected_num_mads;
232 opts.log = true;
233 opts.upper = false;
234 opts.min_diff = -std::log(1 - options.detected_min_drop);
235 host.get_detected() = [&]() {
236 if constexpr(unblocked) {
237 return choose_filter_thresholds::compute(n, res.detected, buffer.data(), opts).lower;
238 } else {
239 return choose_filter_thresholds::internal::strip<true>(choose_filter_thresholds::compute_blocked(n, res.detected, block, &buffer, opts));
240 }
241 }();
242 }
243
244 size_t nsubsets = res.subset_sum.size();
245 host.get_subset_sum().resize(nsubsets);
246 for (size_t s = 0; s < nsubsets; ++s) {
247 auto sub = res.subset_sum[s];
248 choose_filter_thresholds::Options opts;
249 opts.num_mads = options.subset_sum_num_mads;
250 opts.log = true;
251 opts.lower = false;
252 host.get_subset_sum()[s] = [&]() {
253 if constexpr(unblocked) {
254 return choose_filter_thresholds::compute(n, sub, buffer.data(), opts).upper;
255 } else {
256 return choose_filter_thresholds::internal::strip<false>(choose_filter_thresholds::compute_blocked(n, sub, block, &buffer, opts));
257 }
258 }();
259 }

262template<class Host_, typename Sum_, typename Detected_, typename BlockSource_, typename Output_>
263void filter(const Host_& host, size_t n, const MetricsBuffers<Sum_, Detected_>& metrics, BlockSource_ block, Output_* output) {
264 constexpr bool unblocked = std::is_same<BlockSource_, bool>::value;
265 std::fill_n(output, n, 1);
266
267 for (size_t i = 0; i < n; ++i) {
268 auto thresh = [&]() {
269 if constexpr(unblocked) {
270 return host.get_detected();
271 } else {
272 return host.get_detected()[block[i]];
273 }
274 }();
275 output[i] = output[i] && (metrics.detected[i] >= thresh);
276 }
277
278 size_t nsubsets = metrics.subset_sum.size();
279 for (size_t s = 0; s < nsubsets; ++s) {
280 auto sub = metrics.subset_sum[s];
281 const auto& sthresh = host.get_subset_sum()[s];
282 for (size_t i = 0; i < n; ++i) {
283 auto thresh = [&]() {
284 if constexpr(unblocked) {
285 return sthresh;
286 } else {
287 return sthresh[block[i]];
288 }
289 }();
290 output[i] = output[i] && (sub[i] <= thresh);
291 }
292 }
293}
