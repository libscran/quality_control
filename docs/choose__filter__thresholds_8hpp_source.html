77Float_ sanitize(Float_ val, bool unlog) {
78 if (unlog) {
79 if (std::isinf(val)) {

90template<typename Float_>
91std::pair<Float_, Float_> choose(Float_ median, Float_ mad, const Options& options) {
92 static_assert(std::is_floating_point<Float_>::value);
93 Float_ lthresh = -std::numeric_limits<Float_>::infinity();
94 Float_ uthresh = std::numeric_limits<double>::infinity();
95
96 if (!std::isnan(median) && !std::isnan(mad)) {
97 auto delta = std::max(static_cast<Float_>(options.min_diff), options.num_mads * mad);
98 if (options.lower) {
99 lthresh = sanitize(median - delta, options.unlog);
100 }
101 if (options.upper) {
102 uthresh = sanitize(median + delta, options.unlog);
103 }
104 }
105
106 return std::make_pair(lthresh, uthresh);
107}
