125Results<Float_> compute(const find_median_mad::Results<Float_>& mm, const Options& options) {
126 static_assert(std::is_floating_point<Float_>::value);
127 Results<Float_> output;
128 Float_& lthresh = output.lower;
129 Float_& uthresh = output.upper;
130 lthresh = -std::numeric_limits<Float_>::infinity();
131 uthresh = std::numeric_limits<double>::infinity();
132
133 auto median = mm.median;
134 auto mad = mm.mad;
135 if (!std::isnan(median) && !std::isnan(mad)) {
136 auto delta = std::max(static_cast<Float_>(options.min_diff), options.num_mads * mad);
137 if (options.lower) {
138 lthresh = internal::sanitize(median - delta, options.log);
139 }
140 if (options.upper) {
141 uthresh = internal::sanitize(median + delta, options.log);
142 }
143 }
