139Results<Float_> compute(const find_median_mad::Results<Float_>& mm, const Options& options) {
140 static_assert(std::is_floating_point<Float_>::value);
141 Results<Float_> output;
142 Float_& lthresh = output.lower;
143 Float_& uthresh = output.upper;
144 lthresh = -std::numeric_limits<Float_>::infinity();
145 uthresh = std::numeric_limits<double>::infinity();
146
147 auto median = mm.median;
148 auto mad = mm.mad;
149 if (!std::isnan(median) && !std::isnan(mad)) {
150 auto delta = std::max(static_cast<Float_>(options.min_diff), options.num_mads * mad);
151 if (options.lower) {
152 lthresh = internal::sanitize(median - delta, options.log);
153 }
154 if (options.upper) {
155 uthresh = internal::sanitize(median + delta, options.log);
156 }
157 }
158
159 return output;
160}
