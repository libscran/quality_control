72 if (unlog) {

84template<typename Float_>
85std::pair<Float_, Float_> choose(Float_ median, Float_ mad, const Options& options) {
86 static_assert(std::is_floating_point<Float_>::value);
87 Float_ lthresh = -std::numeric_limits<Float_>::infinity();
88 Float_ uthresh = std::numeric_limits<double>::infinity();
89
90 if (!std::isnan(median) && !std::isnan(mad)) {
91 auto delta = std::max(static_cast<Float_>(options.min_diff), options.num_mads * mad);
92 if (options.lower) {
93 lthresh = sanitize(median - delta, options.unlog);
94 }
95 if (options.upper) {
96 uthresh = sanitize(median + delta, options.unlog);
97 }
98 }
99
100 return std::make_pair(lthresh, uthresh);
101}
