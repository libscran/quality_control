85template<typename Index_, typename Float_>
86Results<Float_> compute(Index_ num, Float_* metrics, const Options& options) {
87 static_assert(std::is_floating_point<Float_>::value);
88
89 // Rotate all the NaNs to the front of the buffer and ignore them.
90 Index_ lost = 0;
91 for (Index_ i = 0; i < num; ++i) {
92 if (std::isnan(metrics[i])) {
93 std::swap(metrics[i], metrics[lost]);
94 ++lost;
95 }
96 }
97 metrics += lost;
98 num -= lost;
99
100 if (options.log) {
101 auto copy = metrics;
102 for (Index_ i = 0; i < num; ++i, ++copy) {
103 if (*copy > 0) {
104 *copy = std::log(*copy);
105 } else if (*copy == 0) {
106 *copy = -std::numeric_limits<double>::infinity();
107 } else {
108 throw std::runtime_error("cannot log-transform negative values");
109 }
110 }
111 }
112
113 // No need to skip the NaNs again.
114 auto median = tatami_stats::medians::direct<Float_>(metrics, num, /* skip_nan = */ false);
115
116 if (options.median_only || std::isnan(median)) {
117 // Giving up.
118 return Results<Float_>(median, std::numeric_limits<Float_>::quiet_NaN());
119 } else if (std::isinf(median)) {
120 // MADs should be no-ops when added/subtracted from infinity. Any
121 // finite value will do here, so might as well keep it simple.
122 return Results<Float_>(median, static_cast<Float_>(0));
123 }
124
125 auto copy = metrics;
126 for (Index_ i = 0; i < num; ++i, ++copy) {
127 *copy = std::abs(*copy - median);
128 }
129 auto mad = tatami_stats::medians::direct<Float_>(metrics, num, /* skip_nan = */ false);
130 mad *= 1.4826; // for equivalence with the standard deviation under normality.
131
132 return Results<Float_>(median, mad);
133}

225template<typename Output_ = double, typename Index_ = int, typename Block_ = int, typename Value_ = double>
226std::vector<Results<Output_> > compute_blocked(Index_ num, const Block_* block, const Value_* metrics, Workspace<Output_, Index_>& workspace, const Options& options) {
227 std::vector<Results<Output_> > output;
228
229 auto& buffer = workspace.my_buffer;
230 if (!block) {
231 std::copy_n(metrics, num, buffer.begin());
232 output.push_back(compute(num, buffer.data(), options));
233 return output;
234 }
235
236 const auto& starts = workspace.my_block_starts;
237 auto& ends = workspace.my_block_ends;
238 std::copy(starts.begin(), starts.end(), ends.begin());
239 for (Index_ i = 0; i < num; ++i) {
240 auto& pos = ends[block[i]];
241 buffer[pos] = metrics[i];
242 ++pos;
243 }
244
245 // Using the ranges on the buffer.
246 size_t nblocks = starts.size();
247 output.reserve(nblocks);
248 for (size_t g = 0; g < nblocks; ++g) {
249 output.push_back(compute(ends[g] - starts[g], buffer.data() + starts[g], options));
250 }
251
252 return output;
253}
