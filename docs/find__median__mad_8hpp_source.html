59template<typename Index_, typename Float_>
60std::pair<Float_, Float_> compute(Index_ num, Float_* metrics, const Options& options) {
61 static_assert(std::is_floating_point<Float_>::value);
62
63 // Rotate all the NaNs to the front of the buffer and ignore them.
64 Index_ lost = 0;
65 for (Index_ i = 0; i < num; ++i) {
66 if (std::isnan(metrics[i])) {
67 std::swap(metrics[i], metrics[lost]);
68 ++lost;
69 }
70 }
71 metrics += lost;
72 num -= lost;
73
74 if (options.log) {
75 auto copy = metrics;
76 for (Index_ i = 0; i < num; ++i, ++copy) {
77 if (*copy > 0) {
78 *copy = std::log(*copy);
79 } else if (*copy == 0) {
80 *copy = -std::numeric_limits<double>::infinity();
81 } else {
82 throw std::runtime_error("cannot log-transform negative values");
83 }
84 }
85 }
86
87 // No need to skip the NaNs again.
88 auto median = tatami_stats::medians::direct<Float_>(metrics, num, /* skip_nan = */ false);
89
90 if (options.median_only || std::isnan(median)) {
91 // Giving up.
92 return std::make_pair(median, std::numeric_limits<Float_>::quiet_NaN());
93 } else if (std::isinf(median)) {
94 // MADs should be no-ops when added/subtracted from infinity. Any
95 // finite value will do here, so might as well keep it simple.
96 return std::make_pair(median, static_cast<Float_>(0));
97 }
98
99 auto copy = metrics;
100 for (Index_ i = 0; i < num; ++i, ++copy) {
101 *copy = std::abs(*copy - median);
102 std::cout << *copy << std::endl;
103 }
104 auto mad = tatami_stats::medians::direct<Float_>(metrics, num, /* skip_nan = */ false);
105 std::cout << num << "\t" << mad << std::endl;
106 mad *= 1.4826; // for equivalence with the standard deviation under normality.
107
108 return std::make_pair(median, mad);
109}

201template<typename Output_ = double, typename Index_ = int, typename Block_ = int, typename Value_ = double>
202std::vector<std::pair<Output_, Output_> > compute_blocked(Index_ num, const Block_* block, const Value_* metrics, Workspace<Output_, Index_>& workspace, const Options& options) {
203 std::vector<std::pair<Output_, Output_> > output;
204
205 auto& buffer = workspace.my_buffer;
206 if (!block) {
207 std::copy_n(metrics, num, buffer.begin());
208 output.push_back(compute(num, buffer.data(), options));
209 return output;
210 }
211
212 const auto& starts = workspace.my_block_starts;
213 auto& ends = workspace.my_block_ends;
214 std::copy(starts.begin(), starts.end(), ends.begin());
215 for (Index_ i = 0; i < num; ++i) {
216 auto& pos = ends[block[i]];
217 buffer[pos] = metrics[i];
218 ++pos;
219 }
220
221 // Using the ranges on the buffer.
222 size_t nblocks = starts.size();
223 output.reserve(nblocks);
224 for (size_t g = 0; g < nblocks; ++g) {
225 output.push_back(compute(ends[g] - starts[g], buffer.data() + starts[g], options));
226 }
227
228 return output;
229}
