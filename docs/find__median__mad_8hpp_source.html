59template<typename Value_, typename Index_>
60std::pair<Value_, Value_> compute(Index_ num, Value_* metrics, const Options& options) {
61 // Rotate all the NaNs to the front of the buffer and ignore them.
62 Index_ lost = 0;
63 for (Index_ i = 0; i < num; ++i) {
64 if (std::isnan(metrics[i])) {
65 std::swap(metrics[i], metrics[lost]);
66 ++lost;
67 }
68 }
69 metrics += lost;
70 num -= lost;
71
72 if (options.log) {
73 auto copy = metrics;
74 for (Index_ i = 0; i < num; ++i, ++copy) {
75 if (*copy > 0) {
76 *copy = std::log(*copy);
77 } else if (*copy == 0) {
78 *copy = -std::numeric_limits<double>::infinity();
79 } else {
80 throw std::runtime_error("cannot log-transform negative values");
81 }
82 }
83 }
84
85 // No need to skip the NaNs again.
86 auto median = tatami_stats::medians::direct(metrics, num, /* skip_nan = */ false);
87
88 if (options.median_only || std::isnan(median)) {
89 // Giving up.
90 return std::make_pair(median, std::numeric_limits<Value_>::quiet_NaN());
91 } else if (std::isinf(median)) {
92 // MADs should be no-ops when added/subtracted from infinity. Any
93 // finite value will do here, so might as well keep it simple.
94 return std::make_pair(median, static_cast<Value_>(0));
95 }
96
97 auto copy = metrics;
98 for (Index_ i = 0; i < num; ++i, ++copy) {
99 *copy = std::abs(*copy - median);
100 }
101 auto mad = tatami_stats::medians::direct(metrics, num, /* skip_nan = */ false);
102 mad *= 1.4826; // for equivalence with the standard deviation under normality.
103
104 return std::make_pair(median, mad);
105}

196template<typename Index_, typename Block_, typename Value_>
197std::vector<std::pair<Value_, Value_> > compute_blocked(Index_ num, const Block_* block, const Value_* metrics, Workspace<Value_, Index_>& workspace, const Options& options) {
198 std::vector<std::pair<Value_, Value_> > output;
199
200 auto& buffer = workspace.my_buffer;
201 if (!block) {
202 std::copy_n(metrics, num, buffer.begin());
203 output.push_back(compute(num, buffer.data(), options));
204 return output;
205 }
206
207 const auto& starts = workspace.my_block_starts;
208 auto& ends = workspace.my_block_ends;
209 std::copy(starts.begin(), starts.end(), ends.begin());
210 for (Index_ i = 0; i < num; ++i) {
211 auto& pos = ends[block[i]];
212 buffer[pos] = metrics[i];
213 ++pos;
214 }
215
216 // Using the ranges on the buffer.
217 size_t nblocks = starts.size();
218 output.reserve(nblocks);
219 for (size_t g = 0; g < nblocks; ++g) {
220 output.push_back(compute(ends[g] - starts[g], buffer.data() + starts[g], options));
221 }
222
223 return output;
224}
