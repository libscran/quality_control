85template<typename Index_, typename Float_>
86Results<Float_> compute(Index_ num, Float_* metrics, const Options& options) {
87 static_assert(std::is_floating_point<Float_>::value);
88
89 // Rotate all the NaNs to the front of the buffer and ignore them.
90 Index_ lost = 0;
91 for (Index_ i = 0; i < num; ++i) {
92 if (std::isnan(metrics[i])) {
93 std::swap(metrics[i], metrics[lost]);
94 ++lost;
95 }
96 }
97 metrics += lost;
98 num -= lost;
99
100 if (options.log) {
101 auto copy = metrics;
102 for (Index_ i = 0; i < num; ++i, ++copy) {
103 if (*copy > 0) {
104 *copy = std::log(*copy);
105 } else if (*copy == 0) {
106 *copy = -std::numeric_limits<double>::infinity();
107 } else {
108 throw std::runtime_error("cannot log-transform negative values");
109 }
110 }
111 }
112
113 // No need to skip the NaNs again.
114 auto median = tatami_stats::medians::direct<Float_>(metrics, num, /* skip_nan = */ false);
115
116 if (options.median_only || std::isnan(median)) {
117 // Giving up.
118 return Results<Float_>(median, std::numeric_limits<Float_>::quiet_NaN());
119 } else if (std::isinf(median)) {
120 // MADs should be no-ops when added/subtracted from infinity. Any
121 // finite value will do here, so might as well keep it simple.
122 return Results<Float_>(median, static_cast<Float_>(0));
123 }
124
125 // As an aside, there's no way to avoid passing in 'metrics' as a Float_,
126 // even if the original values were integers, because we need to do this
127 // subtraction here that could cast integers to floats. So at some point we
128 // will need a floating-point buffer, and so we might as well just pass the
129 // metrics in as floats in the first place. Technically the first sort
130 // could be done with an integer buffer but then we'd need an extra argument.
131
132 auto copy = metrics;
133 for (Index_ i = 0; i < num; ++i, ++copy) {
134 *copy = std::abs(*copy - median);
135 }
136 auto mad = tatami_stats::medians::direct<Float_>(metrics, num, /* skip_nan = */ false);
137 mad *= 1.4826; // for equivalence with the standard deviation under normality.
138
139 return Results<Float_>(median, mad);
140}

260std::vector<Results<Output_> > compute_blocked(Index_ num, const Value_* metrics, const Block_* block, Workspace<Output_, Index_>* workspace, const Options& options) {
261 std::unique_ptr<Workspace<Output_, Index_> > xworkspace;
262 if (workspace == NULL) {
263 xworkspace = std::make_unique<Workspace<Output_, Index_> >(num, block);
264 workspace = xworkspace.get();

267 std::vector<Results<Output_> > output;

269 auto& buffer = workspace->my_buffer;
270 if (!block) {

272 output.push_back(compute(num, buffer.data(), options));
273 return output;
274 }
275
276 const auto& starts = workspace->my_block_starts;
277 auto& ends = workspace->my_block_ends;

282 ++pos;

289 output.push_back(compute(ends[g] - starts[g], buffer.data() + starts[g], options));
