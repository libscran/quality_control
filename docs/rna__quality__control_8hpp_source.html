99template<typename Value_, typename Index_, typename Subset_, typename Sum_, typename Detected_, typename Proportion_>
100void compute_metrics(const tatami::Matrix<Value_, Index_>* mat, const std::vector<Subset_>& subsets, MetricsBuffers<Sum_, Detected_, Proportion_>& output, const MetricsOptions& options) {
101 auto NC = mat->ncol();
102 size_t nsubsets = subsets.size();
103
104 per_cell_qc_metrics::Buffers<Sum_, Detected_, Value_, Index_> tmp;
105 tmp.sum = output.sum;
106 tmp.detected = output.detected;
107
108 constexpr bool same_type = std::is_same<Sum_, Proportion_>::value;
109 typename std::conditional<same_type, bool, std::vector<std::vector<Sum_> > >::type placeholder_subset;
110
111 if (output.subset_proportion.size()) {
112 // Providing space for the subset sums if they're not the same type.
113 if constexpr(same_type) {
114 tmp.subset_sum = output.subset_proportion;
115 } else {
116 placeholder_subset.resize(nsubsets);
117 tmp.subset_sum.resize(nsubsets);
118 for (size_t s = 0; s < nsubsets; ++s) {
119 auto& b = placeholder_subset[s];
120 b.resize(NC);
121 tmp.subset_sum[s] = b.data();
122 }
123 }
124 }
125
126 per_cell_qc_metrics::Options opt;
127 opt.num_threads = options.num_threads;
128 per_cell_qc_metrics::compute(mat, subsets, tmp, opt);
129
130 for (size_t s = 0 ; s < nsubsets; ++s) {
131 auto dest = output.subset_proportion[s];
132 if (dest) {
133 auto src = tmp.subset_sum[s];
134 for (Index_ c = 0; c < NC; ++c) {
135 dest[c] = static_cast<Proportion_>(src[c]) / static_cast<Proportion_>(tmp.sum[c]);
136 }
137 }
138 }
139
140 return;
141}

197MetricsResults<Sum_, Detected_, Proportion_> compute_metrics(const tatami::Matrix<Value_, Index_>* mat, const std::vector<Subset_>& subsets, const MetricsOptions& options) {
198 auto NC = mat->ncol();
199 MetricsBuffers<Sum_, Detected_, Proportion_> x;
200 MetricsResults<Sum_, Detected_, Proportion_> output;
201
202 output.sum.resize(NC);
203 x.sum = output.sum.data();
204

206 x.detected = output.detected.data();
207

209 x.subset_proportion.resize(nsubsets);
210 output.subset_proportion.resize(nsubsets);

249void populate(Host_& host, size_t n, const MetricsBuffers<Sum_, Detected_, Proportion_>& res, BlockSource_ block, const FiltersOptions& options) {
250 constexpr bool unblocked = std::is_same<BlockSource_, bool>::value;
251 auto buffer = [&]() {
252 if constexpr(unblocked) {
253 return std::vector<Float_>(n);
254 } else {
255 return find_median_mad::Workspace<Float_, size_t>(n, block);
256 }
257 }();
258
259 {
260 choose_filter_thresholds::Options opts;
261 opts.num_mads = options.sum_num_mads;
262 opts.log = true;
263 opts.upper = false;
264 host.get_sum() = [&]() {
265 if constexpr(unblocked) {
266 return choose_filter_thresholds::compute(n, res.sum, buffer.data(), opts).lower;
267 } else {
268 return choose_filter_thresholds::internal::strip<true>(choose_filter_thresholds::compute_blocked(n, res.sum, block, &buffer, opts));
269 }
270 }();
271 }
272
273 {
274 choose_filter_thresholds::Options opts;
275 opts.num_mads = options.detected_num_mads;
276 opts.log = true;
277 opts.upper = false;
278 host.get_detected() = [&]() {
279 if constexpr(unblocked) {
280 return choose_filter_thresholds::compute(n, res.detected, buffer.data(), opts).lower;
281 } else {
282 return choose_filter_thresholds::internal::strip<true>(choose_filter_thresholds::compute_blocked(n, res.detected, block, &buffer, opts));
283 }
284 }();
285 }
286
287 size_t nsubsets = res.subset_proportion.size();
288 host.get_subset_proportion().resize(nsubsets);
289 for (size_t s = 0; s < nsubsets; ++s) {
290 auto sub = res.subset_proportion[s];
291 choose_filter_thresholds::Options opts;
292 opts.num_mads = options.subset_proportion_num_mads;
293 opts.lower = false;
294 host.get_subset_proportion()[s] = [&]() {
295 if constexpr(unblocked) {
296 return choose_filter_thresholds::compute(n, sub, buffer.data(), opts).upper;
297 } else {
298 return choose_filter_thresholds::internal::strip<false>(choose_filter_thresholds::compute_blocked(n, sub, block, &buffer, opts));
299 }
300 }();
301 }

304template<class Host_, typename Sum_, typename Detected_, typename Proportion_, typename BlockSource_, typename Output_>
305void filter(const Host_& host, size_t n, const MetricsBuffers<Sum_, Detected_, Proportion_>& metrics, BlockSource_ block, Output_* output) {
306 constexpr bool unblocked = std::is_same<BlockSource_, bool>::value;
307 std::fill_n(output, n, 1);
308
309 for (size_t i = 0; i < n; ++i) {
310 auto thresh = [&]() {
311 if constexpr(unblocked) {
312 return host.get_sum();
313 } else {
314 return host.get_sum()[block[i]];
315 }
316 }();
317 output[i] = output[i] && (metrics.sum[i] >= thresh);
318 }
319
320 for (size_t i = 0; i < n; ++i) {
321 auto thresh = [&]() {
322 if constexpr(unblocked) {
323 return host.get_detected();
324 } else {
325 return host.get_detected()[block[i]];
326 }
327 }();
328 output[i] = output[i] && (metrics.detected[i] >= thresh);
329 }
330
331 size_t nsubsets = metrics.subset_proportion.size();
332 for (size_t s = 0; s < nsubsets; ++s) {
333 auto sub = metrics.subset_proportion[s];
334 const auto& sthresh = host.get_subset_proportion()[s];
335 for (size_t i = 0; i < n; ++i) {
336 auto thresh = [&]() {
337 if constexpr(unblocked) {
338 return sthresh;
339 } else {
340 return sthresh[block[i]];
341 }
342 }();
343 output[i] = output[i] && (sub[i] <= thresh);
344 }
345 }
346}
